/*
 * ====================================================================
 * CDS VIEW: ZPRAC_CDS_TABLE_ENTITIES
 * PURPOSE : CDS Table Entities — Defining HANA database tables natively
 *           in CDS without DDIC
 * TOPIC   : define table entity
 * ====================================================================
 *
 * KEY LEARNING POINTS:
 *   1. define table entity is a new CDS DDL statement introduced in
 *      S/4HANA 2025 that creates a HANA database table directly from
 *      CDS source — no separate ABAP Dictionary (DDIC) table artifact
 *      is required as an intermediary.
 *   2. Because the table is defined purely in CDS, the runtime object
 *      IS the HANA table; there is no DDIC table shell to maintain,
 *      reducing double-maintenance and transport complexity.
 *   3. Annotations control every aspect: @EndUserText.label provides
 *      the table description, @AbapCatalog.search.help attaches a
 *      search help, and @AbapCatalog.tableCategory can specify
 *      #TRANSPARENT or #CLUSTER semantics where relevant.
 *   4. Key fields, client handling, data elements, and built-in types
 *      are all supported; the client field (usually client = client;
 *      or using @AbapCatalog.clientDependent) mimics DDIC client
 *      behavior without a separate MANDT column definition.
 *   5. @Semantics annotations for amount, quantity, currency, and
 *      unit of measure work identically to CDS views, enabling the
 *      same analytical and transactional metadata natively on the
 *      table entity.
 *   6. Activation of a define table entity directly generates or
 *      updates the HANA table in the database; there is no separate
 *      activation of a DDIC table followed by a CDS view activation.
 *      Transport is handled via the CDS transport layer only.
 *
 * ====================================================================
 * COMPARISON: Table Entity vs Traditional ABAP Dictionary Table
 * ====================================================================
 * Feature                    | define table entity        | Traditional DDIC Table
 * ---------------------------|----------------------------|---------------------------
 * Artifact type              | CDS source (.asddls)       | DDIC table (.tab)
 * Database object            | Direct HANA table          | Buffered via DDIC layer
 * Separate DDIC step?        | No                         | Yes — maintain table in SE11
 * Transport object           | CDS transport (DDL)        | DDIC transport (R3TR TABL)
 * Annotation support         | Full CDS annotations       | Only DDIC field labels
 * @Semantics.amount          | Yes                        | No (must be defined in view)
 * Client handling            | @AbapCatalog.clientDependent| MANDT field + technical settings
 * Search help attachment     | @AbapCatalog.search.help   | SE11 search help assignment
 * Activation speed           | Single step (CDS → HANA)   | Two steps (DDIC + DB)
 * RAP compatibility          | Native (CDS-managed)       | Requires CDS view wrapper
 *
 * ====================================================================
 * SOURCE CODE: Custom Audit Log Table Entity
 * ====================================================================
 */

@EndUserText.label: 'Custom Audit Log Table Entity'
@AbapCatalog.clientDependent: true
@AbapCatalog.search.help: 'ZPRAC_AUDIT_LOG_SH'
define table entity ZPRAC_AUDIT_LOG_TABLE {

  @EndUserText.label: 'Client'
  client                   : abap.clnt;

  @EndUserText.label: 'Audit ID'
  @AbapCatalog.foreignKey.keyType: #KEY
  key audit_uuid           : sysuuid_x16;

  @EndUserText.label: 'Timestamp'
  @EndUserText.quickInfo: 'UTC Time Stamp of Audit Entry'
  key audit_timestamp      : tzntstmpl;

  @EndUserText.label: 'User Name'
  user_name                : abap.uname;

  @EndUserText.label: 'Action'
  action                   : abap.char(40);

  @EndUserText.label: 'Entity Name'
  entity_name              : abap.char(60);

  @EndUserText.label: 'Entity Key'
  entity_key               : abap.string;

  @EndUserText.label: 'Changed Value (Old)'
  old_value                : abap.string;

  @EndUserText.label: 'Changed Value (New)'
  new_value                : abap.string;

  @EndUserText.label: 'Change Amount'
  @Semantics.amount.currencyCode: 'change_currency'
  change_amount            : abap.curr(15,2);

  @EndUserText.label: 'Change Currency'
  @Semantics.currencyCode: true
  change_currency          : abap.cuky;

  @EndUserText.label: 'Change Quantity'
  @Semantics.quantity.unitOfMeasure: 'change_unit'
  change_quantity          : abap.quan(13,3);

  @EndUserText.label: 'Change Unit'
  @Semantics.unitOfMeasure: true
  change_unit              : abap.unit(3);

}
